// ==UserScript==
// @name         KKTIX auto check
// @namespace    https://kenshinnn.blogspot.tw
// @version      0.1
// @description  自動勾選kktix權利說明
// @author       Kenshinn
// @match        https://kktix.com/events/*
// @grant        none
// ==/UserScript==

(function() {
    'use strict';
    var row = 0; // 第n筆項目
    var count = 2; // 買幾筆
    var time_delayed = 500; // 進入頁面後執行的時間，網速慢時可能需要加大
    var autoClickNext = true;  // 是否自動點擊下一步
    // Your code here...
    setTimeout(function() {
        document.querySelector("#person_agree_terms").click();
        for(var i = 0; i < count; i++) {
            document.querySelectorAll(".ticket-unit .display-table-row button.btn-default.plus")[row].click();
        }
        if(autoClickNext) {
            document.querySelector("button.btn.btn-primary").click();
        }
    }, time_delayed);
})();
